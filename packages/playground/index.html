    console.log("Error: Please enter valid numbers!");

3. Guessing Game

Number guessing game with loops and multiple prompts.

let target = Math.floor(Math.random() * 100) + 1;
let attempts = 0;
let guess;

console.log("I'm thinking of a number between 1 and 100!");

do {
    guess = parseInt(prompt("Attempt " + (attempts + 1) + ": Enter your guess:"));
    attempts++;
    
    if (isNaN(guess)) {
        console.log("Please enter a valid number!");
        continue;
    }
    
    if (guess < target) {
        console.log("Too low! Try higher.");
    } else if (guess > target) {
        console.log("Too high! Try lower.");
    } else {
        console.log("üéâ Congratulations! You guessed it in " + attempts + " attempts!");
        break;
    }
} while (attempts < 7);

if (guess !== target) {
    console.log("Game over! The number was " + target + ".");
}
